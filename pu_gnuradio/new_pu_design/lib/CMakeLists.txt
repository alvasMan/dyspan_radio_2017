########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})
list(APPEND dbconnect_sources
    pktgen_impl.cc
    pktrecv_impl.cc
    pdu_fillpath_cpp_impl.cc
    packet_controller_impl.cc
    cmd_pktgen_impl.cc )

set(dbconnect_sources "${dbconnect_sources}" PARENT_SCOPE)
if(NOT dbconnect_sources)
	MESSAGE(STATUS "No C++ sources... skipping lib/")
	return()
endif(NOT dbconnect_sources)

add_library(gnuradio-dbconnect SHARED ${dbconnect_sources})
target_link_libraries(gnuradio-dbconnect ${Boost_LIBRARIES} ${GNURADIO_ALL_LIBRARIES})
target_link_libraries(gnuradio-dbconnect packet)
set_target_properties(gnuradio-dbconnect PROPERTIES DEFINE_SYMBOL "gnuradio_dbconnect_EXPORTS")

if(APPLE)
    set_target_properties(gnuradio-dbconnect PROPERTIES
        INSTALL_NAME_DIR "${CMAKE_INSTALL_PREFIX}/lib"
    )
endif(APPLE)

########################################################################
# Install built library files
########################################################################
install(TARGETS gnuradio-dbconnect
    LIBRARY DESTINATION lib${LIB_SUFFIX} # .so/.dylib file
    ARCHIVE DESTINATION lib${LIB_SUFFIX} # .lib file
    RUNTIME DESTINATION bin              # .dll file
)

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_dbconnect_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_dbconnect.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_dbconnect.cc
)

add_executable(test-dbconnect ${test_dbconnect_sources})

target_link_libraries(
  test-dbconnect
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-dbconnect
)

GR_ADD_TEST(test_dbconnect test-dbconnect)
